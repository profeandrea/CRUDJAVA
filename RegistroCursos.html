293     public void llenarTablaCursos(String busqueda) {
294         Curso c = new Curso();
295         List<Curso> lista = new ArrayList<Curso>();
296         lista = c.buscarCurso(busqueda);
297         String[][] cursos = new String[lista.size()][9];
298         DefaultTableModel model = new DefaultTableModel();
299         model.addColumn("ID");
300         model.addColumn("Nombre");
301         model.addColumn("Inicio");
302         model.addColumn("Fin");
303         model.addColumn("Dias");
304         model.addColumn("Hora Inicio");
305         model.addColumn("Hora Fin");
306         model.addColumn("Aula");
307         model.addColumn("Descripcion");
308 
309         jtCursos.setModel(model);
310 
311         for (int i = 0; i < lista.size(); i++) {
312             cursos[i][0] = lista.get(i).getIdcurso();
313             cursos[i][1] = lista.get(i).getNombre();
314             cursos[i][2] = lista.get(i).getFechaInicio();
315             cursos[i][3] = lista.get(i).getFechaFin();
316             cursos[i][4] = lista.get(i).getDias();
317             cursos[i][5] = lista.get(i).getHoraInicio();
318             cursos[i][6] = lista.get(i).getHoraFin();
319             cursos[i][7] = lista.get(i).getAula();
320             cursos[i][8] = lista.get(i).getDescripcion();
321             model.addRow(cursos[i]);
322         }
323     }

345     private void jtCursosMouseClicked(java.awt.event.MouseEvent evt) {                                      
346         int fila = jtCursos.getSelectedRow();
347         SimpleDateFormat formato = new SimpleDateFormat("yyyy-MM-dd");
348 
349         String idCurso = jtCursos.getValueAt(fila, 0).toString();
350         String nombre = jtCursos.getValueAt(fila, 1).toString();
351         String fechaInicio = jtCursos.getValueAt(fila, 2).toString();
352         String fechaFin = jtCursos.getValueAt(fila, 3).toString();
353         String dias = jtCursos.getValueAt(fila, 4).toString();
354         String horaInicio = jtCursos.getValueAt(fila, 5).toString();
355         String horaFin = jtCursos.getValueAt(fila, 6).toString();
356         String aula = jtCursos.getValueAt(fila, 7).toString();
357         String descripcion = jtCursos.getValueAt(fila, 8).toString();
358 
359         txtNombre.setText(nombre);
360         txtDescripcion.setText(descripcion);
361         txtAula.setText(aula);
362         try {
363             dcFechaInicio.setDate(formato.parse(fechaInicio));
364             dcFechaFin.setDate(formato.parse(fechaFin));
365         } catch (ParseException ex) {
366             System.out.println("Error parse DAte");
367         }
368         if (dias.equals("LMV")) {
369             cmDias.setSelectedIndex(0);
370         } else if (dias.equals("MJ")) {
371             cmDias.setSelectedIndex(1);
372         } else if (dias.equals("S")) {
373             cmDias.setSelectedIndex(2);
374         } else if (dias.equals("D")) {
375             cmDias.setSelectedIndex(3);
376         } else {
377             cmDias.setSelectedIndex(1);
378         }
379         txtHoraInicio.setText(horaInicio);
380         txtHoraFin.setText(horaFin);
381     }

383     private void btnModificarMouseClicked(java.awt.event.MouseEvent evt) {                                          
384         int resultado = 0;
385 
386         Curso curso = new Curso();
387 
388         SimpleDateFormat formato = new SimpleDateFormat("yyyy-MM-dd");
389         //Capturar datos de los campos
390         String fechaInicio = formato.format(dcFechaInicio.getDate());
391         String fechaFin = formato.format(dcFechaFin.getDate());
392 
393         String nombre = txtNombre.getText();
394         String descripcion = txtDescripcion.getText();
395 
396         String aula = txtAula.getText();
397         String horaInicio = txtHoraInicio.getText();
398         String horaFin = txtHoraFin.getText();
399         String idCurso = jtCursos.getValueAt(jtCursos.getSelectedRow(), 0).toString();
400 
401         String dias = "";
402 
403         if (cmDias.getSelectedIndex() == 0) {
404             dias = "LMV";
405         } else if (cmDias.getSelectedIndex() == 1) {
406             dias = "MJ";
407         } else if (cmDias.getSelectedIndex() == 2) {
408             dias = "S";
409         } else if (cmDias.getSelectedIndex() == 3) {
410             dias = "D";
411         } else {
412             //Mostrar mensaje de completar todos los campos
413         }
414 
415         curso.setAula(aula);
416         curso.setIdcurso(idCurso);
417         curso.setDescripcion(descripcion);
418         curso.setDias(dias);
419         curso.setFechaFin(fechaFin);
420         curso.setFechaInicio(fechaInicio);
421         curso.setHoraInicio(horaInicio);
422         curso.setHoraFin(horaFin);
423         curso.setNombre(nombre);
424 
425         resultado = curso.modificarCurso();
426         if (resultado > 0) {
427             llenarTablaCursos("");
428             limpiar();
429             JOptionPane.showMessageDialog(rootPane, "Curso Modificado");
430         } else {
431             JOptionPane.showMessageDialog(rootPane, "Curso no modificado, intente de nuevo");
432         }
