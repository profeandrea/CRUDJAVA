117     private void jButton1MouseClicked(java.awt.event.MouseEvent evt) {                                      
118         //Capturar usuario y contrase√±a
119         String idUsuario, contra;
120         idUsuario = txtUsuario.getText();
121         contra = String.valueOf(txtContra.getPassword());
122         //Creamos un objeto usuario
123         Usuario nuevo = new Usuario();
124         nuevo.setIdUsuario(idUsuario);
125         nuevo.setContra(contra);
126         //Llamamos al metodo autenticar
127         Usuario retorno = new Usuario();
128         retorno = nuevo.autenticar();
129         
130         if(retorno != null){
131             JOptionPane.showMessageDialog(rootPane, "Bienvenido " + retorno.getIdUsuario());
132             Principal form = new Principal();
133             this.setVisible(false);
134             form.setVisible(true);
135         }else{
136             JOptionPane.showMessageDialog(rootPane, "Usuario y/o Contrase√±a no validos");
137         }
138     }
