127     private void btnEstudiantesMouseClicked(java.awt.event.MouseEvent evt) {                                            
128         RegistroEstudiantes registroE = new RegistroEstudiantes();
129         //Quitamos la seleccion de otros botones, menos el que se ha presionado
130         btnCursos.setSelected(false);
131         btnInscripciones.setSelected(false);
132         btnInicio.setSelected(false);
133         
134         panelDesktop.removeAll();
135 //Vamos a cargar el formulario de registro de estudiantes en el desktoppane
136         panelDesktop.add(registroE).setVisible(true);
137         try {
138             //Aqui le decimos al formulario que tome el tamaño maximo del desktoppane
139             registroE.setMaximum(true);
140         } catch (Exception e) {
141             System.out.println("Error al setMaximun registro E");
142         }
143 
144     }
