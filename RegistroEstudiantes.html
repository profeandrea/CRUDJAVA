299     private void llenarTablaEstudiantes(String busqueda) {
300         Estudiante e = new Estudiante();
301         List<Estudiante> lista = new ArrayList<Estudiante>();
302         lista = e.buscarEstudiante(busqueda);
303         String[][] estudiantes = new String[lista.size()][7];
304         DefaultTableModel model = new DefaultTableModel();
305         model.addColumn("ID");
306         model.addColumn("Nombres");
307         model.addColumn("Apellidos");
308         model.addColumn("Email");
309         model.addColumn("Fecha de nacimiento");
310         model.addColumn("Género");
311         model.addColumn("Dirección");
312         jtEstudiantes.setModel(model);
313 
314         for (int i = 0; i < lista.size(); i++) {
315             estudiantes[i][0] = lista.get(i).getIdestudiante();
316             estudiantes[i][1] = lista.get(i).getNombres();
317             estudiantes[i][2] = lista.get(i).getApellidos();
318             estudiantes[i][3] = lista.get(i).getEmail();
319             estudiantes[i][4] = lista.get(i).getFechaNac();
320             estudiantes[i][5] = lista.get(i).getGenero();
321             estudiantes[i][6] = lista.get(i).getDireccion();
322 
323             model.addRow(estudiantes[i]);
324         }
325 
326     }

336     private void btnGuardarMouseClicked(java.awt.event.MouseEvent evt) {                                        
337         //Capturar datos
338         String nombres, apellidos, idEstudiante, fechaNac, email, direccion, genero;
339         SimpleDateFormat formato = new SimpleDateFormat("yyyy-MM-dd");
340         Estudiante e = new Estudiante();
341         int filas = 0;
342         nombres = txtNombres.getText();
343         apellidos = txtApellidos.getText();
344         email = txtEmail.getText();
345         direccion = txtDireccion.getText();
346         fechaNac = formato.format(dcFechaNacimiento.getDate());
347 
348         if (rbFemenino.isSelected()) {
349             genero = "F";
350         } else {
351             genero = "M";
352         }
353         idEstudiante = nombres.substring(0, 1) + apellidos.substring(0, 1) + fechaNac.substring(0, 4);
354 
355         e.setNombres(nombres);
356         e.setApellidos(apellidos);
357         e.setDireccion(direccion);
358         e.setGenero(genero);
359         e.setFechaNac(fechaNac);
360         e.setEmail(email);
361         e.setIdestudiante(idEstudiante);
362         filas = e.agregarEstudiante();
363         if (filas > 0) {
364             JOptionPane.showMessageDialog(rootPane, "Estudiante Agregado");
365             limpiar();
366             llenarTablaEstudiantes("");
367         } else {
368             JOptionPane.showMessageDialog(rootPane, "Estudiante NO Agregado, intente de nuevo");
369         }
370     }

372     private void jtEstudiantesMouseClicked(java.awt.event.MouseEvent evt) {                                           
373         int fila = jtEstudiantes.getSelectedRow();
374         SimpleDateFormat formato = new SimpleDateFormat("yyyy-MM-dd");
375         String fechaNac = jtEstudiantes.getValueAt(fila, 4).toString();
376         try {
377             dcFechaNacimiento.setDate(formato.parse(fechaNac));
378         } catch (ParseException ex) {
379             System.out.println("Exception al convertir fecha");
380         }
381         txtIdEstudiante.setText(jtEstudiantes.getValueAt(fila, 0).toString());
382         txtNombres.setText(jtEstudiantes.getValueAt(fila, 1).toString());
383         txtApellidos.setText(jtEstudiantes.getValueAt(fila, 2).toString());
384         txtEmail.setText(jtEstudiantes.getValueAt(fila, 3).toString());
385         txtDireccion.setText(jtEstudiantes.getValueAt(fila, 6).toString());
386         String genero = jtEstudiantes.getValueAt(fila, 5).toString();
387         if (genero.equals("F")) {
388             rbFemenino.setSelected(true);
389         } else {
390             rbMasculino.setSelected(true);
391         }
